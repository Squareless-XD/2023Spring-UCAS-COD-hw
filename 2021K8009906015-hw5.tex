\problem{6.20} 用原码一位乘、两位乘和补码一位乘 (Booth算法) 、两位乘计算$x·y$.

\begin{enumerate}
    \item $x=  0.110 111$ ,  $y= -0.101 110$ ;
    
    \item $x= -0.010 111$ ,  $y= -0.010 101$ ;
    
    \item $x=  19$        ,  $y=  35$        ;
    
    \item $x=  0.110 11$  ,  $y= -0.111 01$  .
\end{enumerate}
